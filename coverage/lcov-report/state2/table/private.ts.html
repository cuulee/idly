export type Table<T> = Map<string, T>;
 
export const tableCreate = <T>(): Table<T> => new Map<string, T>();
 
export const tableAdd = <T>(t: Table<T>, k: string, e: T) => t.set(k, e);
 
export const tableRemove = <T>(t: Table<T>, k: string) => {
  t.delete(k);
  return t;
};

export const tableGet = <T>(t: Table<T>, k: string) => t.get(k);

export const tableHas = <T>(t: Table<T>, k: string) => t.has(k);
 
export const tableMap = <T>(f: (r: T) => T, t: Table<T>) => {
  const result: T[] = [];
  t.forEach(e => f(e));
  return result;
};

export const tableFilter = <T>(
  f: (r: T, id: string) => boolean,
  t: Table<T>
) => {
  const result: T[] = [];
  t.forEach((e, id) => f(e, id) && result.push(e));
  return result;
};
 
export type tableFindType = <T>(
  f: (r: T, id: string) => boolean,
  t: Table<T>
) => T | undefined;
 
export const tableFind: tableFindType = (f, t) => {
  for (const [id, e] of t) {
    if (f(e, id)) {
      return e;
    }
  }
  return;
};
