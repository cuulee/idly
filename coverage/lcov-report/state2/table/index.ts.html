import { Identity } from 'monet';
import { weakCache } from '../../misc/weakCache';
import { Entity } from '../../osm/structures';
import { mapFilterKeys } from '../helper';
import {
  Table,
  tableAdd as _tableAdd,
  tableGet as _tableGet,
  tableRemove as _tableRemove,
} from './private';
export type EntityTable = Table<Entity>;
 
export type EntityTableGetType = (
  eTable: EntityTable,
  key: string
) => Entity | undefined;

export const entityTableGet: EntityTableGetType = (eTable, key) =>
  _tableGet(eTable, key);
 
export const entityTableCreate = (): EntityTable => new Map();
 
export const entityTableAddBulk = (
  eTable: EntityTable,
  entities: Entity[],
  quadkey = ''
) =>
  Identity(eTable)
    .map(t => quadkeyTableAddBulk(t, entities, quadkey))
    .map(t => entities.reduce((prev, e) => _tableAdd(prev, e.id, e), t))
    .get();
 
export const entityTableRemoveBulk = (eTable: EntityTable, keys: string[]) =>
  Identity(eTable)
    .map(t => quadkeyTableRemoveBulk(t, keys))
    .map(t => keys.reduce((prev, e) => _tableRemove(prev, e), t))
    .get();
 
export type EntityTableFindByQuadkeyType = (
  eTable: EntityTable,
  quadkey: string
) => Entity[];
 
export const entityTableFindByQuadkey: EntityTableFindByQuadkeyType = (
  eTable,
  quadkey
) =>
  quadkeyTableFind(eTable, quadkey)
    .map(k => _tableGet(eTable, k))
    .filter(r => r) as Entity[];
 
export const quadkeyTableFind = (eTable: EntityTable, quadkey: string) =>
  mapFilterKeys(v => v.startsWith(quadkey), getLinkedQuadkeyTable(eTable));
 
export const getLinkedQuadkeyTable = weakCache(
  // tslint:disable-next-line:variable-name
  (_table: EntityTable): Map<string, string> => new Map()
);
 
export const quadkeyTableAddBulk = (
  eTable: EntityTable,
  entities: Entity[],
  quadkey: string
) => {
  const qTable = getLinkedQuadkeyTable(eTable);
  entities.reduce((prev, e) => _tableAdd(prev, e.id, quadkey), qTable);
  return eTable;
};
 
export const quadkeyTableRemoveBulk = (eTable: EntityTable, keys: string[]) => {
  const qTable = getLinkedQuadkeyTable(eTable);
  keys.reduce((prev, e) => _tableRemove(prev, e), qTable);
  return eTable;
};
