import { Either, Left, Right } from 'monet';
import { Entity, EntityType, Node, Relation, Way } from '../osm/structures';
import { isVirgin } from './helper';
import { getId, getLatestVersion, getVersion, Log } from './log';
import { EntityTable } from './table';
export const checkId = (obj: any) => obj.hasOwnProperty('id');

export const checkTypeNode = (obj: any) => checkType(obj, EntityType.NODE);

export const checkTypeWay = (obj: any) => checkType(obj, EntityType.WAY);

export const checkTypeRelation = (obj: any) =>
  checkType(obj, EntityType.RELATION);
 
export const checkType = (obj: any, type: string) =>
  obj.hasOwnProperty('type') && obj.type === type;

export const checkLonLat = (obj: any) =>
  obj.hasOwnProperty('loc') &&
  typeof obj.loc.lat === 'number' &&
  typeof obj.loc.lon === 'number';

export const checkTags = (obj: any) =>
  obj.hasOwnProperty('tags') &&
  Object.keys(obj.tags)
    .map(k => obj.tags[k])
    .every(v => typeof v === 'string');
 
export const checkAttributes = (obj: any) =>
  obj.hasOwnProperty('attributes') &&
  Object.keys(obj.attributes)
    .map(k => (obj.attributes as any)[k])
    .every(v => typeof v === 'string' || typeof v === 'boolean');

export const checkValidNode = (obj: any): obj is Node =>
  obj &&
  checkId(obj) &&
  checkTypeNode(obj) &&
  checkLonLat(obj) &&
  checkTags(obj) &&
  checkAttributes(obj);

export const check = (obj: Entity) => {
  if (obj.type === EntityType.NODE) {
    return checkValidNode(obj);
  }
  if (obj.type === EntityType.WAY) {
    return checkValidWay(obj);
  }
  if (obj.type === EntityType.RELATION) {
    return checkValidRelation(obj);
  }
  return;
};
export const checkNd = (obj: any) =>
  Array.isArray(obj.nodes) && obj.nodes.every(n => typeof n === 'string');

export const checkMembers = (obj: Relation) =>
  obj.hasOwnProperty('members') &&
  obj.members.every(m => m.hasOwnProperty('id'));

export const checkValidWay = (obj: any): obj is Way =>
  obj &&
  checkId(obj) &&
  checkTypeWay(obj) &&
  checkTags(obj) &&
  checkAttributes(obj) &&
  checkNd(obj);

export const checkValidRelation = (obj: any): obj is Relation =>
  obj &&
  checkId(obj) &&
  checkTypeRelation(obj) &&
  checkTags(obj) &&
  checkAttributes(obj) &&
  checkMembers(obj);

export const getParent = (indexOrId: string, eTable: EntityTable) => {
  if (isVirgin(indexOrId)) {
    return undefined;
  }
  const version = getVersion(indexOrId);
  if (version === 0) {
    return eTable.get(getId(indexOrId));
  }
  return eTable.get(`${getId(indexOrId)}#${version - 1}`);
};
 
export const validateEntity = (
  log: Log,
  eTable: EntityTable,
  entities: Entity[]
): Either<Error, Log> => {
  for (const e of entities) {
    if (!check(e)) {
      return Left(new Error('malformed entity ' + e.id));
    }
 
    if (isVirgin(e.id)) {
      return Left(new Error('the passed id is virgin ' + e.id));
    }
 
    if (!getParent(e.id, eTable)) {
      return Left(new Error('couldnt find parent of ' + e.id));
    }
 
    if (getLatestVersion(e.id)(log) + 1 !== getVersion(e.id)) {
      return Left(
        new Error(
          `The version of ${e.id} should be +1 of the lastest version in log.`
        )
      );
    }
  }
  return Right(log);
};
