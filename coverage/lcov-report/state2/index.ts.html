import { Left, Maybe, Right, Nothing, Some } from 'monet';
import * as R from 'ramda';
import { Entity } from '../osm/structures';
import { addEntryToLog, getId, getLatestVersion, Log } from './log';
import {
  EntityTable,
  entityTableAddBulk,
  entityTableGet,
  entityTableRemoveBulk,
  quadkeyTableFind,
} from './table';
import { validateEntity } from './validation';
 
/**
 * The log is used as [...modifiedEntries, virginOsmEntries]
 * the pointer to virginOsmEntries resolves to EntityTable
 * we carefully mutate virginOsmEntries keeping the pointer same.
 *
 * modified entities are id#version where version [0, infinity) (0 is inclusive)
 */
 
export const commitModifiedEntities = (log: Log, eTable: EntityTable) => (
  entities: Entity[]
) => {
  return Right<Error, Log>(log)
    .flatMap(l => validateEntity(log, eTable, entities))
    .map(l => addEntryToLog(new Set([...entities.map(r => r.id)]))(l))
    .flatMap(
      l =>
        l === log
          ? Left<Error, Log>(new Error('Log already has some of the entities'))
          : Right<Error, Log>(l)
    )
    .map(l => R.tap(() => entityTableAddBulk(eTable, entities), l))
    .cata(err => {
      throw err;
    }, set => set);
};
 
export function getLatestEntity(
  log: Log,
  eTable: EntityTable,
  indexOrId: string
): Maybe<Entity> {
  const version = getLatestVersion(indexOrId)(log);
  return MaybeOf(
    entityTableGet(eTable, version === -1 ? getId(indexOrId) : indexOrId)
  );
}
 
const MaybeOf = <T>(a: T | undefined) => (a ? Some<T>(a) : Nothing<T>());
 
export function getEntityByIndex(
  eTable: EntityTable,
  index: string
): Maybe<Entity> {
  return MaybeOf(eTable.get(index));
}
 
export const addVirginEntities = (
  eTable: EntityTable,
  virginEntities: Entity[],
  quadkey: string
) => entityTableAddBulk(eTable, virginEntities, quadkey);
 
export const removeVirginEntities = (
  eTable: EntityTable,
  virginIds: string[]
) => entityTableRemoveBulk(eTable, virginIds);
 
export const getEntitiesInQuadkey = (
  log: Log,
  eTable: EntityTable,
  quadkey: string
) => {
  return quadkeyTableFind(eTable, quadkey);
};
